apt-get update -y && apt-get upgrade -y

pkg install -y git nodejs ffmpeg yarn

git clone https://github.com/G4tito/Simple-WaBot.git

# Navigate to the project directory
cd Simple-WaBot

# Install project dependencies
yarn install

# Run the application
node .
